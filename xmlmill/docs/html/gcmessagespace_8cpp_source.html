00157   bool userAccepted( const QString &uniqueMessageKey,
00158                      const QString &heading,
00159                      const QString &text,
00160                      ButtonCombo buttons,
00161                      Buttons defaultButton,
00162                      Icon icon,
00163                      bool saveCancel )
00164   {
00165     if( !settingsInitialised )

00167       settings.setValue( "dialogPreferences", "" );

00170 
00171     /* Check if the user previously requested that his/her choice must be saved. */
00172     QString key = QString( "dialogPreferences/%1" ).arg( uniqueMessageKey );
00173     QString valueKey = QString( key + "/preference" );
00174 
00175     bool remembered = settings.value( key, false ).toBool();
00176 
00177     if( !remembered )
00178     {
00179       GCMessageDialog dialog( &remembered, heading, text, buttons, defaultButton, icon );
00180       QDialog::DialogCode accept = static_cast< QDialog::DialogCode >( dialog.exec() );
00181 
00182       if( accept == QDialog::Accepted )
00183       {
00184         if( remembered )
00185         {
00186           settings.setValue( key, true );
00187           settings.setValue( valueKey, true );
00188         }
00189 
00190         return true;
00191       }
00192       else
00193       {
00194         /* For some message prompts, it isn't desirable to save "Cancel" values and
00195           it is therefore necessary to ensure that none of these values are remembered
00196           by accident. */
00197         if( remembered )
00198         {
00199           if( saveCancel )
00200           {
00201             settings.setValue( key, true );
00202             settings.setValue( valueKey, false );
00203           }
00204           else
00205           {
00206             settings.setValue( key, false );
00207           }
00208         }
00209 
00210         return false;
00211       }
00212     }
00213     else
00214     {
00215       /* If we do have a remembered setting, act accordingly. */
00216       return settings.value( valueKey ).toBool();
