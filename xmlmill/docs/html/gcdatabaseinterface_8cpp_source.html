00307 {
00308   if( element.isEmpty() )
00309   {
00310     m_lastErrorMsg = QString( "Trying to add an empty element name." );
00311     return false;
00312   }
00313 
00314   QSqlQuery query = selectElement( element );
00315 
00316   /* If we don't have an existing record, add it. */
00317   if( !query.first() )

00319     if( !query.prepare( INSERT_ELEMENT ) )
00320     {
00321       m_lastErrorMsg = QString( "Prepare INSERT element failed for element \"%1\": [%2]" )

00324       return false;
00325     }
00326 
00327     query.addBindValue( element );
00328     query.addBindValue( cleanAndJoinListElements( children ) );
00329     query.addBindValue( cleanAndJoinListElements( attributes ) );
00330 
00331     if( !query.exec() )

00333       m_lastErrorMsg = QString( "INSERT element failed for element \"%1\": [%2]" )

00747 bool GCDataBaseInterface::isDocumentCompatible( const QDomDocument *doc ) const
00748 {
00749   GCBatchProcessorHelper helper( doc,
00750                                  SEPARATOR,
00751                                  knownElements(),
00752                                  knownAttributeKeys() );
00753 
00754   qApp->processEvents( QEventLoop::ExcludeUserInputEvents );
00755 
00756   /* If there are any new elements or attributes to add, the document is incompatible (not checking
00757     for new attribute values since new values don't affect XML relationships, i.e. it isn't important
00758     enough to import entire documents each time an unknown value is encountered) */
00759   if( !helper.newAssociatedElementsToAdd().isEmpty() ||
00760       !helper.newAttributeKeysToAdd().isEmpty() ||
00761       !helper.newElementAttributesToAdd().isEmpty() ||
00762       !helper.newElementChildrenToAdd().isEmpty() ||
00763       !helper.newElementsToAdd().isEmpty() )
00764   {
00765     return false;
00766   }
00767 
00768   /* If there are no new items, we may still have previously unknown relationships which we
00769     need to check.  This is slightly more involved than simply checking for empty lists.
00770     Also remember that the lists returned by GCBatchProcessorHelper are all synchronised
00771     with regards to their indices (which is the only reason why we can loop through the
00772     lists like this). */
00773   for( int i = 0; i < helper.elementsToUpdate().size(); ++i )
00774   {
00775     /* If any new element children were added, we have an incompatible document. */
00776     QStringList knownChildren = children( helper.elementsToUpdate().at( i ).toString() );
00777     QStringList allChildren = QStringList() << knownChildren << helper.elementChildrenToUpdate().at( i ).toString().split( SEPARATOR );
00778     cleanList( allChildren );
00779 
00780     /* Check for larger since we only care about added items. */
00781     if( allChildren.size() > knownChildren.size() )
00782     {
00783       return false;
00784     }
00785 
00786     /* If any new attributes were added, we also have an incompatible document. */
00787     QStringList knownAttributes = attributes( helper.elementsToUpdate().at( i ).toString() );
00788     QStringList allAttributes = QStringList() << knownAttributes << helper.elementAttributesToUpdate().at( i ).toString().split( SEPARATOR );
00789     cleanList( allAttributes );
00790 
00791     if( allAttributes.size() > knownAttributes.size() )
00792     {
00793       return false;
00794     }
00795   }
00796 
00797   return true;
