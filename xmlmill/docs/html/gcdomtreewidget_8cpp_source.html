00193 int GCDomTreeWidget::findItemPositionAmongDuplicates( const QString &nodeText, int itemIndex ) const
00194 {
00195   QList< int > indices;
00196 
00197   if( !m_isEmpty )
00198   {
00199     /* If there are multiple nodes with the same element name (more likely than not), check which
00200     of these nodes are exact duplicates with regards to attributes, values, etc. */
00201     for( int i = 0; i < m_items.size(); ++i )
00202     {
00203       GCTreeWidgetItem *treeItem = m_items.at( i );
00204 
00205       if( treeItem->toString() == nodeText )
00206       {
00207         indices.append( treeItem->index() );
00208       }
00209     }
00210   }
00211 
00212   /* Now that we have a list of all the indices matching identical nodes (indices are a rough
00213     indication of an element's position in the DOM and closely matches the "line numbers" of the
00214     items in the tree widget), we can determine the position of the selected DOM element relative
00215     to its doppelgangers and highlight its text representation in the text edit area. */
00216   qSort( indices.begin(), indices.end() );
00217 
00218   return indices.indexOf( itemIndex );
00219 }

00328 void GCDomTreeWidget::replaceItemsWithComment( const QList< int > &indices, const QString &comment )
00329 {  
00330   QList< GCTreeWidgetItem* > itemsToDelete;
00331   GCTreeWidgetItem *commentParentItem = NULL;
00332 
00333   for( int i = 0; i < m_items.size(); ++i )
00334   {
00335     GCTreeWidgetItem *item = m_items.at( i );
00336 
00337     for( int j = 0; j < indices.size(); ++j )
00338     {
00339       if( item->index() == indices.at( j ) )
00340       {
00341         /* This works because the indices are always sorted from small to big, i.e.
00342           the item corresponding to the lowest index in indices will be the furthest up
00343           the node hierarchy. */
00344         if( j == 0 && item->gcParent() )
00345         {
00346           commentParentItem = item->gcParent();
00347         }
00348 
00349         /* Remove the element from the DOM first. */
00350         QDomNode parentNode = item->element().parentNode();
00351         parentNode.removeChild( item->element() );
00352 
00353         /* Now whack it. */
00354         if( item->gcParent() )
00355         {
00356           GCTreeWidgetItem *parentItem = item->gcParent();
00357           parentItem->removeChild( item );
00358         }
00359         else
00360         {
00361           invisibleRootItem()->removeChild( item );
00362         }
00363 
00364         /* Removing an item from another's child list does not delete it. */
00365         itemsToDelete.append( item );
00366       }
00367     }
00368   }
00369 
00370   /* Delete the items here so that we may be sure that they are actually removed
00371     from the items list as well (deleting items in the loop above resulted in parent
00372     items deleting all their children, but obviously not updating the items list in
00373     the process). */
00374   for( int i = 0; i < itemsToDelete.size(); ++ i )
00375   {
00376     GCTreeWidgetItem *item = itemsToDelete.at( i );
00377     removeFromList( item );
00378     delete item;
00379     item = NULL;
00380   }
00381 
00382   /* Create a comment node with the combined text of all the item nodes that were removed
00383     and insert it in the correct position in the DOM document. */
00384   QDomComment newComment = m_domDoc->createComment( comment );
00385 
00386   if( commentParentItem )
00387   {
00388     commentParentItem->element().appendChild( newComment );
00389   }
00390   else
00391   {
00392     m_domDoc->appendChild( newComment );
00393   }
00394 
00395   m_comments.append( newComment );
00396   m_isEmpty = m_items.isEmpty();
00397   updateIndices();
00398 }

00448 void GCDomTreeWidget::processNextElementFromDatabase( const QString &element )
00449 {
00450   QStringList children = GCDataBaseInterface::instance()->children( element );
00451 
00452   foreach( QString child, children )
00453   {
00454     addItem( child );
00455 
00456     /* Since it isn't illegal to have elements with children of the same name, we cannot
00457       block it in the DB, however, if we DO have elements with children of the same name,
00458       this recursive call enters an infinite loop, so we need to make sure that doesn't
00459       happen. */
00460     GCTreeWidgetItem *newItem = gcCurrentItem();
00461 
00462     if( !parentTreeAlreadyContainsElement( newItem, child ) )
00463     {
00464       processNextElementFromDatabase( child );
00465     }
00466     else
00467     {
00468       setCurrentItem( m_activeItem->parent() );  // required to enforce sibling relationships
00469     }
00470   }
00471 
00472   setCurrentItem( m_activeItem->parent() );  // required to enforce sibling relationships
00473 }

00516 void GCDomTreeWidget::insertItem( const QString &elementName, int index, bool toParent )
00517 {
00518   QDomElement element = m_domDoc->createElement( elementName );
00519 
00520   /* Create all the possible attributes for the element here, they can be changed
00521     later on. */
00522   QStringList attributeNames = GCDataBaseInterface::instance()->attributes( elementName );
00523 
00524   for( int i = 0; i < attributeNames.count(); ++i )
00525   {
00526     element.setAttribute( attributeNames.at( i ), "" );
00527   }
00528 
00529   GCTreeWidgetItem *item = new GCTreeWidgetItem( element, m_items.size() );
00530   m_items.append( item );
00531 
00532   if( m_isEmpty )
00533   {
00534     invisibleRootItem()->addChild( item );  // takes ownership
00535     m_domDoc->appendChild( element );
00536     m_isEmpty = false;
00537   }
00538   else
00539   {
00540     if( !toParent )
00541     {
00542       m_activeItem->insertGcChild( index, item );
00543     }
00544     else
00545     {
00546       m_activeItem->gcParent()->insertGcChild( index, item );
00547     }
00548   }
00549 
00550   /* I will have to rethink this approach if it turns out that it is too expensive to
00551     iterate through the tree on each and every addition...for now, this is the easiest
00552     solution, even if not the best. */
00553   updateIndices();
00554 
00555   setCurrentItem( item );
00556 }

00683 void GCDomTreeWidget::dropEvent( QDropEvent *event )
00684 {
00685   m_itemBeingManipulated = true;
00686 
00687   QTreeWidget::dropEvent( event );
00688   DropIndicatorPosition indicatorPos = dropIndicatorPosition();
00689 
00690   if( m_activeItem )
00691   {
00692     bool moveComment = false;
00693 
00694     if( m_activeItem->element().previousSibling().isComment() )
00695     {
00696       m_commentNode = m_activeItem->element().previousSibling().toComment();
00697       moveComment = true;
00698     }
00699 
00700     QDomElement previousParent = m_activeItem->element().parentNode().toElement();
00701     previousParent.removeChild( m_activeItem->element() );
00702 
00703     GCTreeWidgetItem *parent = m_activeItem->gcParent();
00704 
00705     if( parent )
00706     {
00707       if( indicatorPos == QAbstractItemView::OnItem )
00708       {
00709         parent->element().appendChild( m_activeItem->element() );
00710       }
00711       else if( indicatorPos == QAbstractItemView::AboveItem ||
00712                indicatorPos == QAbstractItemView::BelowItem )
00713       {
00714         GCTreeWidgetItem *sibling = NULL;
00715         int pos = parent->indexOfChild( m_activeItem );
00716 
00717         if( pos > 0 )
00718         {
00719           sibling = parent->gcChild( pos - 1 );
00720 
00721           if( sibling )
00722           {
00723             parent->element().insertAfter( m_activeItem->element(), sibling->element() );
00724           }
00725         }
00726         else if( pos == 0 )
00727         {
00728           sibling = parent->gcChild( pos + 1 );
00729 
00730           if( sibling )
00731           {
00732             parent->element().insertBefore( m_activeItem->element(), sibling->element() );
00733           }
00734         }
00735         else
00736         {
00737           parent->element().appendChild( m_activeItem->element() );
00738         }
00739       }
00740 
00741       /* Move the associated comment (if any). */

00830 {
00831   QString newName = QInputDialog::getText( this, "Change element name", "Enter the element's new name:" );
00832 
00833   if( !newName.isEmpty() && m_activeItem )

00838

00841        add the element if it doesn't.  In the latter case, the children  and attributes associated with
00842        the old name will be assigned to the new element in the process. */
00843     QStringList attributes = GCDataBaseInterface::instance()->attributes( oldName );

00846     if( !GCDataBaseInterface::instance()->addElement( newName, children, attributes ) )
00847     {
00848       GCMessageSpace::showErrorMessageBox( this, GCDataBaseInterface::instance()->lastError() );
00849     }

00851     /* If we are, in fact, dealing with a new element, we also want the "new" element's associated attributes
00852       to be updated with the known values of these attributes. */
00853     foreach( QString attribute, attributes )
00854     {
00855       QStringList attributeValues = GCDataBaseInterface::instance()->attributeValues( oldName, attribute );
00856 
00857       if( !GCDataBaseInterface::instance()->updateAttributeValues( newName, attribute, attributeValues ) )
00858       {
00859         GCMessageSpace::showErrorMessageBox( this, GCDataBaseInterface::instance()->lastError() );

00869 void GCDomTreeWidget::removeItem()
00870 {
00871   if( m_activeItem )
00872   {
00873     m_itemBeingManipulated = true;
00874 
00875     /* I think it is safe to assume that comment nodes will exist just above an element
00876       although it might not always be the case that a multi-line comment exists within
00877       a single set of comment tags.  However, for those cases, it's the user's responsibility
00878       to clean them up. */
00879     if( !m_commentNode.isNull() )
00880     {
00881       /* Check if the comment is an actual comment or if it's valid XML that's been
00882         commented out (we don't want to remove snippets). */
00883       QDomDocument doc;
00884 
00885       if( !doc.setContent( m_commentNode.nodeValue() ) )
00886       {
00887         m_comments.removeAll( m_commentNode );
00888         m_commentNode.parentNode().removeChild( m_commentNode );
00889       }
00890     }
00891 
00892     /* Remove the element from the DOM first. */
00893     QDomNode parentNode = m_activeItem->element().parentNode();
00894     parentNode.removeChild( m_activeItem->element() );
00895 
00896     /* Now whack it. */
00897     if( m_activeItem->gcParent() )
00898     {
00899       GCTreeWidgetItem *parentItem = m_activeItem->gcParent();
00900       parentItem->removeChild( m_activeItem );
00901     }
00902     else
00903     {
00904       invisibleRootItem()->removeChild( m_activeItem );
00905     }
00906 
00907     removeFromList( m_activeItem );
00908     m_isEmpty = m_items.isEmpty();
00909 
00910     delete m_activeItem;
00911     m_activeItem = gcCurrentItem();
00912 
00913     updateIndices();
00914     emitGcCurrentItemChanged( m_activeItem, 0 );
00915     m_itemBeingManipulated = false;
00916   }
00917 }

00938 
00939         /* Update the database to reflect the re-parenting. */
00940         GCDataBaseInterface::instance()->updateElementChildren( grandParent->name(), QStringList( m_activeItem->name() ) );
00941       }

00953 void GCDomTreeWidget::stepDown()
00954 {
00955   if( m_activeItem )
00956   {
00957     m_itemBeingManipulated = true;
00958 
00959     GCTreeWidgetItem* parentItem = m_activeItem->gcParent();
00960     GCTreeWidgetItem* siblingItem = gcItemFromNode( m_activeItem->element().previousSiblingElement() );
00961 
00962     /* Try again in the opposite direction. */
00963     if( !siblingItem )
00964     {
00965       siblingItem = gcItemFromNode( m_activeItem->element().nextSiblingElement( ) );
00966     }
00967 
00968     if( siblingItem && parentItem )
00969     {
00970       parentItem->removeChild( m_activeItem );
00971       siblingItem->insertChild( 0, m_activeItem );
00972       siblingItem->element().insertBefore( m_activeItem->element(), siblingItem->element().firstChild() );
00973 
00974       /* Update the database to reflect the re-parenting. */
00975       GCDataBaseInterface::instance()->updateElementChildren( siblingItem->name(), QStringList( m_activeItem->name() ) );
00976 
00977       updateIndices();
00978       emitGcCurrentItemChanged( m_activeItem, 0 );
00979     }
00980
