00227   QList< GCTreeWidgetItem* > includedItems;
00228   ui->treeWidget->getIncludedTreeWidgetItems( includedItems );
00229 
00230   /* Add the required number of snippets. */
00231   for( int i = 0; i < ui->spinBox->value(); ++i )
00232   {
00233     /* Update all the included elements and attribute values. */
00234     for( int j = 0; j < includedItems.size(); ++j )
00235     {
00236       GCTreeWidgetItem* localItem = includedItems.at( j );
00237 
00238       /* Sets a "restore point" so that we may increment attribute values and return
00239         to the previously fixed values (to avoid incrementing an incremented value). */
00240       localItem->fixAttributeValues();
00241 
00242       QString elementName = localItem->element().tagName();
00243       QDomNamedNodeMap attributes = localItem->element().attributes();
00244 
00245       for( int k = 0; k < attributes.size(); ++k )
00246       {
00247         QDomAttr attr = attributes.item( k ).toAttr();
00248         QString attributeValue = localItem->fixedValue( attr.name() );
00249 
00250         if( localItem->incrementAttribute( attr.name() ) )
00251         {
00252           /* Check if this is a number (if it contains any non-digit character). */
00253           if( !attributeValue.contains( QRegExp( "\\D+" ) ) )
00254           {
00255             bool ok( false );
00256             int intValue = attributeValue.toInt( &ok );
00257 
00258             if( ok )
00259             {
00260               intValue += i;
00261               attributeValue = QString( "%1" ).arg( intValue );
00262             }
00263           }
00264           else
00265           {
00266             /* If the value contains some string characters, it's a string value and that's all
00267               there is to it (it's not our responsibility to check that someone isn't incrementing
00268               "false", e.g.). */
00269             attributeValue += QString( "%1" ).arg( i );
00270           }
00271 
00272           localItem->element().setAttribute( attr.name(), attributeValue );
00273         }
00274 
00275         /* This call does nothing if the attribute value already exists. */
00276         GCDataBaseInterface::instance()->updateAttributeValues( elementName, attr.name(), QStringList( attributeValue ) );
00277       }      
00278     }
00279 
00280     emit snippetAdded( m_parentItem, ui->treeWidget->cloneDocument().toElement() );
00281 
00282     /* Restore values. */
00283     for( int j = 0; j < includedItems.size(); ++j )

00285       includedItems.at( j )->revertToFixedValues();

00292 void GCAddSnippetsForm::updateCheckStates( GCTreeWidgetItem *item ) const
00293 {
00294   /* Checking or unchecking an item must recursively update its children as well. */
00295   if( item->checkState( 0 ) == Qt::Checked )
00296   {
00297     item->setExcludeElement( false );
00298 
00299     /* When a low-level child is activated, we need to also update its parent tree all the way
00300       up to the root element since including a child automatically implies that the parent
00301       element is included. */
00302     GCTreeWidgetItem *parent = item->gcParent();
00303 
00304     while( parent && ( parent->checkState( 0 ) != Qt::Checked ) )
00305     {
00306       parent->setExcludeElement( false );
00307       parent->setCheckState( 0, Qt::Checked );
00308       parent = parent->gcParent();
00309     }
00310   }
00311   else
00312   {
00313     item->setExcludeElement( true );
00314 
00315     for( int i = 0; i < item->childCount(); ++i )
00316     {
00317       item->child( i )->setCheckState( 0, item->checkState( 0 ) );
00318       updateCheckStates( item->gcChild( i ) );
00319     }
00320   }
00321 }
