00125 void GCRemoveItemsForm::deleteElement( const QString &element )
00126 {
00127   /* If the element name is empty, then this function was called directly by the user
00128     clicking on "delete" (as opposed to this function being called further down below
00129     during the recursive process of getting rid of the element's children) in that case,
00130     the first element to be removed is the current one (set in "elementSelected"). */
00131   QString currentElement = ( element.isEmpty() ) ? m_currentElement : element;
00132 
00133   QStringList children = GCDataBaseInterface::instance()->children( currentElement );
00134   m_deletedElements.clear();
00135 
00136   /* Attributes and values must be removed before we can remove elements and we must also
00137     ensure that children are removed before their parents.  To achieve this, we need to ensure
00138     that we clean the element tree from "the bottom up". */
00139   if( !children.isEmpty() )
00140   {
00141     foreach( QString child, children )
00142     {
00143       if( GCDataBaseInterface::instance()->isUniqueChildElement( currentElement, child ) )
00144       {
00145         deleteElement( child );
00146       }
00147     }
00148   }
00149 
00150   /* Remove all the attributes (and their known values) associated with this element. */
00151   QStringList attributes = GCDataBaseInterface::instance()->attributes( currentElement );
00152 
00153   foreach( QString attribute, attributes )
00154   {
00155     if( !GCDataBaseInterface::instance()->removeAttribute( currentElement, attribute ) )
00156     {
00157       GCMessageSpace::showErrorMessageBox( this, GCDataBaseInterface::instance()->lastError() );
00158     }
00159   }
00160 
00161   /* Now we can remove the element itself. */
00162   if( !GCDataBaseInterface::instance()->removeElement( currentElement ) )
00163   {
00164     GCMessageSpace::showErrorMessageBox( this, GCDataBaseInterface::instance()->lastError() );
00165   }
00166   else
00167   {
00168     m_deletedElements.append( currentElement );
00169   }
00170 
00171   /* Check if the user removed a root element. */
00172   if( GCDataBaseInterface::instance()->knownRootElements().contains( currentElement ) )
00173   {
00174     if( !GCDataBaseInterface::instance()->removeRootElement( currentElement ) )
00175     {
00176       GCMessageSpace::showErrorMessageBox( this, GCDataBaseInterface::instance()->lastError() );
00177     }
00178   }
00179 
00180   /* Remove the element from all its parents' first level child lists. */
00181   updateChildLists();
00182 
00183   ui->comboBox->clear();
00184   ui->plainTextEdit->clear();
00185   ui->treeWidget->populateFromDatabase();
00186 }

00207                                                       GCMessageSpace::YesNo,

00210

00283   QStringList knownElements = GCDataBaseInterface::instance()->knownElements();
00284 
00285   foreach( QString element, knownElements )
00286   {
00287     QStringList children = GCDataBaseInterface::instance()->children( element );
00288 
00289     if( !children.isEmpty() )
00290     {
00291       foreach( QString deletedElement, m_deletedElements )
00292       {
00293         if( children.contains( deletedElement ) )
00294         {
00295           if( !GCDataBaseInterface::instance()->removeChildElement( element, deletedElement ) )
00296           {

00299         }
