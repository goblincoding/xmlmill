00258       GCComboBox *attributeCombo = new GCComboBox;

00299

00359 void GCMainWindow::attributeChanged( QTableWidgetItem *tableItem )
00360 {
00361   /* Don't execute the logic if a tree widget item's activation is triggering
00362     a re-population of the table widget (which results in this slot being called). */
00363   if( !m_wasTreeItemActivated && !m_newAttributeAdded )
00364   {
00365     /* Also don't allow for empty attribute names. */
00366     if( tableItem->text().isEmpty() )
00367     {
00368       tableItem->setText( m_activeAttributeName );
00369       return;
00370     }
00371 
00372     /* When the check state of a table widget item changes, the "itemChanged" signal is emitted
00373       before the "itemClicked" one and this messes up the logic that follows completely.  Since
00374       I depend on knowing which attribute is currently active, I needed to insert the following
00375       odd little check (the other alternative is probably to subclass QTableWidgetItem but I'm
00376       worried about the added complications). */
00377     if( m_activeAttribute != tableItem )
00378     {
00379       m_activeAttributeName = tableItem->text();
00380     }
00381 
00382     /* All attribute name changes will be assumed to be additions, removing an attribute
00383       with a specific name has to be done explicitly. */
00384     GCTreeWidgetItem *treeItem = ui->treeWidget->gcCurrentItem();
00385 
00386     /* See if an existing attribute's name changed or if a new attribute was added. */
00387     if( tableItem->text() != m_activeAttributeName )
00388     {
00389       /* Add the new attribute's name to the current element's list of associated attributes. */
00390       if( GCDataBaseInterface::instance()->updateElementAttributes( treeItem->name(), QStringList( tableItem->text() ) ) )
00391       {
00392         /* Is this a name change? */
00393         if( m_activeAttributeName != EMPTY )
00394         {
00395           treeItem->excludeAttribute( m_activeAttributeName );
00396 
00397           /* Retrieve the list of values associated with the previous name and insert it against the new name. */
00398           QStringList attributeValues = GCDataBaseInterface::instance()->attributeValues( treeItem->name(),
00399                                                                                           m_activeAttributeName );
00400 
00401           if( !GCDataBaseInterface::instance()->updateAttributeValues( treeItem->name(), tableItem->text(), attributeValues ) )
00402           {
00403             GCMessageSpace::showErrorMessageBox( this, GCDataBaseInterface::instance()->lastError() );
00404           }
00405         }
00406         else
00407         {
00408           /* If this is an entirely new attribute, insert an "empty" row so that the user may add
00409             even more attributes if he/she wishes to do so. */
00410           m_newAttributeAdded = true;
00411           tableItem->setFlags( tableItem->flags() | Qt::ItemIsUserCheckable );
00412           tableItem->setCheckState( Qt::Checked );
00413           insertEmptyTableRow();
00414           m_newAttributeAdded = false;
00415         }
00416       }
00417       else
00418       {
00419         GCMessageSpace::showErrorMessageBox( this, GCDataBaseInterface::instance()->lastError() );
00420       }
00421     }
00422 
00423     m_activeAttributeName = tableItem->text();
00424 
00425     /* Is this attribute included or excluded? */
00426     GCComboBox *attributeValueCombo = dynamic_cast< GCComboBox* >( ui->tableWidget->cellWidget( tableItem->row(), VALUESCOLUMN ) );
00427 
00428     if( tableItem->checkState() == Qt::Checked )
00429     {
00430       attributeValueCombo->setEnabled( true );
00431       treeItem->includeAttribute( m_activeAttributeName, attributeValueCombo->currentText() );

00433     else
00434     {
00435       attributeValueCombo->setEnabled( false );

00492   querySetActiveSession( QString( "No active profile set, please set one for this session." ) );

00495   {

00498 
00499   QString fileName = QFileDialog::getOpenFileName( this, "Open File", QDir::homePath(), "XML Files (*.*)" );
00500 
00501   /* If the user cancelled, we don't want to continue. */
00502   if( fileName.isEmpty() )
00503   {
00504     return false;
00505   }
00506 
00507   /* Note to future self: although the user would have explicitly saved (or not saved) the file
00508     by the time this functionality is encountered, we only reset the document once we have a new,
00509     active file to work with since users are fickle and may still change their minds.  In other
00510     words, do NOT move this code to before "getOpenFileName" as previously considered! */
00511   resetDOM();

00513

00515

00517   {

00519                        .arg( fileName )
00520                        .arg( file.errorString() );
00521     GCMessageSpace::showErrorMessageBox( this, errorMsg );
00522     return false;
00523   }
00524 
00525   QTextStream inStream( &file );
00526   QString fileContent( inStream.readAll() );
00527   qint64 fileSize = file.size();
00528   file.close();
00529 
00530   /* This application isn't optimised for dealing with very large XML files (the entire point is that

00541   }

00543   {
00544     QMessageBox::critical( this,
00545                            "Very large file!",
00546                            "This file is too large to edit manually!" );
00547 
00548     return false;
00549   }
00550 
00551   QString xmlErr( "" );
00552   int     line  ( -1 );
00553   int     col   ( -1 );
00554 
00555   if( !ui->treeWidget->setContent( fileContent, &xmlErr, &line, &col ) )
00556   {
00557     QString errorMsg = QString( "XML is broken - Error [%1], line [%2], column [%3]" )
00558                        .arg( xmlErr )
00559                        .arg( line )
00560                        .arg( col );

00562     resetDOM();

00574     {

00901 
00902 void GCMainWindow::addElementToDocument()
00903 {
00904   QString elementName = ui->addElementComboBox->currentText();
00905   bool treeWasEmpty = ui->treeWidget->isEmpty();
00906   bool addToParent = false;
00907 
00908   /* If the user selected the <element> option, we add a new sibling element
00909     of the same name as the current element to the current element's parent. */
00910   if( elementName.contains( QRegExp( LEFTRIGHTBRACKETS ) ) )
00911   {
00912     elementName = elementName.remove( QRegExp( LEFTRIGHTBRACKETS ) );
00913     addToParent = true;
00914   }
00915 
00916   /* There is probably no chance of this ever happening, but defensive programming FTW! */
00917   if( !elementName.isEmpty() )
00918   {
00919     /* Update the tree widget. */
00920     ui->treeWidget->addItem( elementName, addToParent );
00921 
00922     GCTreeWidgetItem *treeItem = ui->treeWidget->gcCurrentItem();
00923     treeItem->setFlags( treeItem->flags() | Qt::ItemIsEditable );
00924 
00925     /* If the user starts creating a DOM document without having explicitly asked for
00926     a new file to be created, do it automatically (we can't call "newXMLFile here" since
00927     it resets the DOM document). */
00928     if( treeWasEmpty )
00929     {      
00930       m_currentXMLFileName = "";
00931       ui->actionCloseFile->setEnabled( true );
00932       ui->actionSave->setEnabled( true );
00933       ui->actionSaveAs->setEnabled( true );
00934       ui->addSnippetButton->setEnabled( true );
00935       ui->addChildElementButton->setText( "Add Child" );
00936     }
00937 
00938     /* Add all the known attributes associated with this element name to the new element. */
00939     QStringList attributes = GCDataBaseInterface::instance()->attributes( elementName );
00940 
00941     for( int i = 0; i < attributes.size(); ++i )
00942     {
00943       treeItem->element().setAttribute( attributes.at( i ), QString( "" ) );
00944     }
00945 
00946     setTextEditContent( treeItem );
00947     elementSelected( treeItem, 0 );
00948   }
00949 }
