00045 void removeDuplicates( QList< int > &indices )
00046 {
00047   for( int i = 0; i < indices.size(); ++i )
00048   {
00049     if( indices.count( indices.at( i ) ) > 1 )
00050     {
00051       int backup = indices.at( i );
00052 
00053       /* Remove all duplicates. */
00054       indices.removeAll( backup );
00055 
00056       /* Add one occurrence back. */
00057       indices.append( backup );
00058     }
00059   }
00060 }

00175 void GCPlainTextEdit::commentOutSelection()
00176 {
00177   m_cursorPositionChanging = true;
00178 
00179   /* Capture the text before we make any changes. */
00180   QString comment = textCursor().selectedText();
00181 
00182   int selectionStart = textCursor().selectionStart();
00183   int selectionEnd = textCursor().selectionEnd();
00184 
00185   QTextCursor cursor = textCursor();
00186   cursor.setPosition( selectionEnd );
00187   cursor.movePosition( QTextCursor::EndOfBlock );
00188 
00189   int finalBlockNumber = cursor.blockNumber();
00190 
00191   cursor.setPosition( selectionStart );
00192   cursor.movePosition( QTextCursor::StartOfBlock );
00193 
00194   QList< int > indices;
00195   QTextBlock block = cursor.block();
00196 
00197   while( block.isValid() &&
00198          block.blockNumber() <= finalBlockNumber )
00199   {
00200     indices.append( findIndexMatchingBlockNumber( block ) );
00201     block = block.next();
00202   }
00203 
00204   cursor.setPosition( selectionStart );
00205   cursor.beginEditBlock();
00206   cursor.insertText( OPENCOMMENT );
00207   cursor.endEditBlock();
00208 
00209   cursor.setPosition( selectionEnd );
00210   cursor.movePosition( QTextCursor::EndOfBlock );
00211   cursor.beginEditBlock();
00212   cursor.insertText( CLOSECOMMENT );
00213   cursor.endEditBlock();
00214 
00215   setTextCursor( cursor );
00216 
00217   if( confirmDomNotBroken( 2 ) )
00218   {
00219     comment = comment.replace( QChar( 0x2029 ), '\n' );    // replace Unicode end of line character
00220     comment = comment.trimmed();
00221     removeDuplicates( indices );
00222     emit commentOut( indices, comment );
00223   }
00224 
00225   m_cursorPositionChanging = false;
00226 }

00313 bool GCPlainTextEdit::confirmDomNotBroken( int undoCount )
00314 {
00315   QString xmlErr( "" );
00316   int     line  ( -1 );
00317   int     col   ( -1 );
00318 
00319   /* Create a temporary document so that we do not mess with the contents
00320     of the tree item node map and current DOM if the new XML is broken. */
00321   QDomDocument doc;
00322 
00323   if( !doc.setContent( toPlainText(), &xmlErr, &line, &col ) )
00324   {
00325     /* Unfortunately the line number returned by the DOM doc doesn't match up with what's
00326       visible in the QTextEdit.  It seems as if it's mostly off by one line.  For now it's a
00327       fix, but will have to figure out how to make sure that we highlight the correct lines.
00328       Ultimately this finds the broken XML and highlights it in red...what a mission... */
00329     QTextBlock textBlock = document()->findBlockByLineNumber( line - 1 );
00330     QTextCursor cursor( textBlock );
00331     cursor.movePosition( QTextCursor::NextWord );
00332     cursor.movePosition( QTextCursor::EndOfBlock, QTextCursor::KeepAnchor );
00333 
00334     m_savedPalette = cursor.blockCharFormat().background();
00335 
00336     QTextEdit::ExtraSelection highlight;
00337     highlight.cursor = cursor;
00338     highlight.format.setBackground( QColor( 220, 150, 220 ) );
00339     highlight.format.setProperty  ( QTextFormat::FullWidthSelection, true );
00340 
00341     QList< QTextEdit::ExtraSelection > extras;
00342     extras << highlight;
00343     setExtraSelections( extras );
00344     ensureCursorVisible();
00345 
00346     QString errorMsg = QString( "XML is broken - Error [%1], line [%2], column [%3].\n\n"
00347                                 "Your action will be reverted." )
00348                        .arg( xmlErr )
00349                        .arg( line )
00350                        .arg( col );
00351     GCMessageSpace::showErrorMessageBox( this, errorMsg );
00352 
00353     for( int i = 0; i < undoCount; ++i )
00354     {
00355       undo();
00356     }
00357 
00358     highlight.cursor = textCursor();
00359     highlight.format.setBackground( m_savedPalette );
00360     highlight.format.setProperty  ( QTextFormat::FullWidthSelection, true );
00361 
00362     extras.clear();
00363     extras << highlight;
00364     setExtraSelections( extras );

00366   }

00373 int GCPlainTextEdit::findIndexMatchingBlockNumber( QTextBlock block )
00374 {
00375   int itemNumber = block.blockNumber();
00376   int errorCounter = 0;
00377   bool insideComment = false;
00378 
00379   while( block.isValid() &&
00380          block.blockNumber() >= 0 )
00381   {
00382     /* Check if we just entered a comment block (this is NOT wrong, remember
00383       that we are working our way back up the document, not down). */
00384     if( block.text().contains( CLOSECOMMENT ) )
00385     {
00386       errorCounter = 0;
00387       insideComment = true;
00388     }
00389 
00390     if( insideComment ||
00391         block.text().contains( "</" ) ||          // element close
00392         block.text().remove( " " ).isEmpty() ||   // empty lines
00393         ( block.text().contains( "<?" ) &&
00394           block.text().contains( "?>" ) ) )       // xml version specification
00395     {
00396       itemNumber--;
00397     }
00398 
00399     /* Check if we are about to exit a comment block. */
00400     if( block.text().contains( OPENCOMMENT ) )
00401     {
00402       /* If we are exiting but we never entered, then we need to compensate for the
00403         subtractions we've done erroneously. */
00404       if( !insideComment )
00405       {
00406         itemNumber -= errorCounter;
00407       }
00408 
00409       insideComment = false;
00410     }
00411 
00412     errorCounter++;
00413     block = block.previous();
00414   }
00415 
00416   return itemNumber;
00417 }
